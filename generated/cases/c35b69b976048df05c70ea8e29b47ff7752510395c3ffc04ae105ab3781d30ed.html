Project Context
This milestone uses the public replication dataset forTalk or Type? The Effect of Digital Interfaces on
Citizens’ Satisfaction with Standardized Public Services.
I’ve researched the extent to which individuals have had previous experiences with on-line delivery of services.
In this proposal I am evaluating if individuals are more satisfied with how public services are delivered via
internet vs. face-to-face means. This script documents all data preparation, primary independent variables,
and exploration of preliminary statistical analyses.

Cleaning Strategy
The survey uses three repeated vignette blocks. Each block contains:
•the requested service type,
•the communication channel,
•the service quality scenario,
•procedure satisfaction, and
•result satisfaction.
I first filter to fully completed responses, then reshape the vignette information into a long format with one
row per vignette.
parse_service_type <-function(x) {
dplyr::case_when(
str_detect(x,regex("Reisepass", ignore_case = TRUE))~"Passport",
str_detect(x,regex("Führungszeugnis", ignore_case = TRUE))~"Certificate of good conduct",
str_detect(x,regex("Wohnberechtigungsschein", ignore_case = TRUE))~"Social housing",
TRUE~NA_character_
)
}
parse_channel <-function(x) {
dplyr::case_when(
str_detect(x,regex("Bürgerservice-App", ignore_case = TRUE))~"App",
str_detect(x,regex("Self-Service-Terminal", ignore_case = TRUE))~"Self-service terminal",
str_detect(x,regex("Sachbearbeiter", ignore_case = TRUE))~"Face-to-face",
TRUE~NA_character_
)
}
parse_quality <-function(x) {
dplyr::case_when(
str_detect(x,regex("Systemfehler", ignore_case = TRUE))~"Failure with explanation",
2
str_detect(x,regex("entschuldigt", ignore_case = TRUE))~"Failure with explanation",
str_detect(x,regex("erst am nächsten Tag abholen können", ignore_case = TRUE))~"Failure without explanation",
str_detect(x,regex("und holen", ignore_case = TRUE))~"No failure",
TRUE~NA_character_
)
}
recode_digital_experience <-function(app, terminal) {
dplyr::case_when(
app==1|terminal==1~"Yes",
app==2&terminal==2~"No",
TRUE~NA_character_
)
}
make_long_block <-function(data, period) {
service_var <-paste0("verwaltungsvorgang", period)
channel_var <-paste0("komebene", period)
quality_var <-paste0("serviceebene", period)
proc_var <-paste0("vign", period, "1")
result_var <-paste0("vign", period, "2")
data%>%
transmute(
respondent_id =row_number(),
period = period,
age =as.numeric(alter),
gender =factor(
geschlecht,
levels =c(1, 2),
labels =c("Male", "Female")
),
income_group =factor(
eink,
levels =c(1, 2, 3),
labels =c("Low", "Middle", "High")
),
prior_app =as.integer(erfahrungapp==1),
prior_terminal =as.integer(erfahrungsst==1),
prior_digital_experience =recode_digital_experience(erfahrungapp, erfahrungsst),
service_type =parse_service_type(.data[[service_var]]),
channel =parse_channel(.data[[channel_var]]),
service_quality =parse_quality(.data[[quality_var]]),
satisfaction_process =as.numeric(.data[[proc_var]]),
satisfaction_result =as.numeric(.data[[result_var]]),
satisfaction =rowMeans(cbind(
as.numeric(.data[[proc_var]]),
as.numeric(.data[[result_var]])
), na.rm = TRUE)
)
}
